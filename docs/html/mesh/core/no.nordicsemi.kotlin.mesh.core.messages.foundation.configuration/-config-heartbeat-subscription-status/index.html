ConfigHeartbeatSubscriptionStatus

class ConfigHeartbeatSubscriptionStatus(val status: ConfigMessageStatus, val source: HeartbeatSubscriptionSource, val destination: HeartbeatSubscriptionDestination, val periodLog: PeriodLog, val countLog: CountLog, val minHops: UByte, val maxHops: UByte) : ConfigResponse, ConfigStatusMessage
